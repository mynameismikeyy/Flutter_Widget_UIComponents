import 'package:flutter/material.dart';
import 'package:flutter_staggered_grid_view/flutter_staggered_grid_view.dart';

class HomePage extends StatelessWidget {
  final List<String> images = [
    "https://picsum.photos/300/500?random=1",
    "https://picsum.photos/300/400?random=2",
    "https://picsum.photos/300/600?random=3",
    "https://picsum.photos/300/450?random=4",
    "https://picsum.photos/300/550?random=5",
    "https://picsum.photos/300/350?random=6",
    "https://picsum.photos/300/700?random=7",
    "https://picsum.photos/300/300?random=8",
  ];

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      backgroundColor: Colors.grey[100], // CHANGED: light background
      appBar: AppBar(
        title: Text("Pinterest Layout"), // CHANGED: custom title
        centerTitle: true,               // CHANGED: title nasa gitna
        elevation: 0,                    // CHANGED: no shadow appbar
        backgroundColor: Colors.redAccent, // CHANGED: ibang kulay appbar
      ),
      body: Padding(
        padding: const EdgeInsets.all(8.0), // CHANGED: may padding sa paligid
        child: MasonryGridView.count(
          crossAxisCount: 2, // CHANGED: 2-column Pinterest style
          mainAxisSpacing: 10, // CHANGED: vertical spacing
          crossAxisSpacing: 10, // CHANGED: horizontal spacing
          itemCount: images.length,
          itemBuilder: (context, index) {
            return PinterestCard(
              imageUrl: images[index],
              title: "Item ${index + 1}", // CHANGED: dynamic title
            );
          },
        ),
      ),
    );
  }
}

class PinterestCard extends StatelessWidget {
  final String imageUrl;
  final String title;

  const PinterestCard({required this.imageUrl, required this.title});

  @override
  Widget build(BuildContext context) {
    return Material(
      elevation: 6, // CHANGED: may shadow yung card
      borderRadius: BorderRadius.circular(16), // CHANGED: rounded corners
      clipBehavior: Clip.hardEdge, // CHANGED: para hindi lumampas yung image sa corners
      child: Stack(
        children: [
          // Background Image
          Image.network(
            imageUrl,
            fit: BoxFit.cover, // CHANGED: cover instead of default
          ),
          // Gradient Overlay
          Positioned.fill(
            child: Container(
              decoration: BoxDecoration(
                gradient: LinearGradient( // CHANGED: gradient overlay
                  colors: [
                    Colors.black.withOpacity(0.5),
                    Colors.transparent,
                  ],
                  begin: Alignment.bottomCenter,
                  end: Alignment.topCenter,
                ),
              ),
            ),
          ),
          // Title Text
          Positioned(
            bottom: 10,
            left: 10,
            child: Text(
              title,
              style: TextStyle(
                color: Colors.white, // CHANGED: white text para kita sa image
                fontSize: 16,        // CHANGED: mas malaking font
                fontWeight: FontWeight.bold, // CHANGED: bold style
              ),
            ),
          )
        ],
      ),
    );
  }
}
